// GAME STATE let gameData = { email: '', cafecoin: 1000, litecoin: 0, miningPower: 0, machines: [], inventory: [], currentRoom: 1, maxRacks: 6 }; // MÁQUINAS DISPONÍVEIS const SHOP_ITEMS = { 'Miner Cafeteira': { price: 50, hash: 5, name: '☕ Miner Cafeteira' }, 'Miner Croissant': { price: 120, hash: 12, name: '🥐 Miner Croissant' }, 'Miner Espresso Pro': { price: 300, hash: 30, name: '⚡ Miner Espresso Pro' } }; // INICIALIZAÇÃO document.addEventListener('DOMContentLoaded', function() { loadGame(); setupEventListeners(); updateDisplay(); checkLogin(); }); function setupEventListeners() { // Navegação document.querySelectorAll('.nav-btn').forEach((btn, index) => { btn.onclick = () => switchTab(index); }); // Botões carteira document.getElementById('deposit-btn').onclick = depositLTC; document.getElementById('withdraw-btn').onclick = openWithdrawModal; document.getElementById('confirm-withdraw-btn').onclick = confirmWithdraw; document.getElementById('close-modal-btn').onclick = closeWithdrawModal; document.getElementById('buy-room-btn').onclick = buyRoom; } function checkLogin() { if (!gameData.email) { document.getElementById('login-screen').style.display = 'flex'; } else { document.getElementById('login-screen').style.display = 'none'; } } function handleAuth() { const email = document.getElementById('email-input').value; const pass = document.getElementById('password-input').value; if (email && pass) { gameData.email = email; saveGame(); document.getElementById('login-screen').style.display = 'none'; updateDisplay(); } } function handleLogout() { if (confirm('Sair do jogo?')) { localStorage.removeItem('cafeMinerSave'); location.reload(); } } function switchTab(index) { document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('active')); document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active')); document.querySelectorAll('.nav-btn')[index].classList.add('active'); document.querySelectorAll('.tab-content')[index].classList.add('active'); } function buyItem(name, price, hash) { if (gameData.cafecoin >= price) { gameData.cafecoin -= price; gameData.inventory.push({ name: name, hash: hash, id: Date.now() }); updateDisplay(); saveGame(); alert(`${name} comprado! Vá pro Inventário pra instalar.`); } else { alert('CaféCoins insuficiente!'); } updateInventory(); } function updateInventory() { const list = document.getElementById('inventory-list'); if (gameData.inventory.length === 0) { list.innerHTML = '

Seu inventário está vazio. Visite a loja!

'; return; } list.innerHTML = ''; gameData.inventory.forEach((item, index) => { const card = document.createElement('div'); card.className = 'shop-card'; card.innerHTML = `

${item.name}

+${item.hash} H/s

`; list.appendChild(card); }); } function installMachine(index) { if (gameData.machines.length >= gameData.maxRacks) { alert('Sala lotada! Compre nova sala.'); return; } const machine = gameData.inventory.splice(index, 1)[0]; gameData.machines.push(machine); gameData.miningPower += machine.hash; updateDisplay(); updateInventory(); updateRacks(); saveGame(); } function updateRacks() { for (let i = 1; i <= 6; i++) { for (let j = 1; j <= 3; j++) { const shelf = document.getElementById(`shelf-${i}-${j}`); if (shelf) shelf.innerHTML = '
Vazio
'; } } gameData.machines.forEach((machine, index) => { const row = Math.floor(index / 3) + 1; const col = (index % 3) + 1; const shelf = document.getElementById(`shelf-${row}-${col}`); if (shelf) { shelf.innerHTML = `
${machine.name}
${machine.hash} H/s
`; } }); // Timer banner const banner = document.getElementById('timer-banner'); if (gameData.machines.length > 0) { banner.innerHTML = '✅ SISTEMA OPERANDO: Minerando Litecoin...'; banner.style.background = '#1e3a2f'; } else { banner.innerHTML = '⚠️ SISTEMA DESLIGADO: Coloque uma máquina para iniciar a mineração.'; banner.style.background = '#3a1e1e'; } } function buyRoom() { const price = 500; if (gameData.cafecoin >= price) { gameData.cafecoin -= price; gameData.maxRacks += 6; gameData.currentRoom++; document.getElementById('room-name-display').innerText = `SALA ${gameData.currentRoom} (1/3)`; updateDisplay(); saveGame(); alert('Nova sala comprada! +6 slots'); } else { alert('CaféCoins insuficiente! Precisa de 500 CC'); } } function depositLTC() { const valor = prompt('Quantos LTC deseja depositar na carteira do jogo?'); if (valor &&!isNaN(valor) && parseFloat(valor) > 0) { gameData.litecoin += parseFloat(valor); updateDisplay(); saveGame(); alert(`${valor} LTC depositados!`); } } function openWithdrawModal() { document.getElementById('modal-available-ltc').innerText = gameData.litecoin.toFixed(8); document.getElementById('withdraw-modal').style.display = 'flex'; } function closeWithdrawModal() { document.getElementById('withdraw-modal').style.display = 'none'; } function confirmWithdraw() { const endereco = document.getElementById('wallet-input').value; const valor = parseFloat(document.getElementById('withdraw-amount-input').value); if (!endereco || endereco.length < 10) { alert('Endereço LTC inválido!'); return; } if (!valor || valor < 0.001) { alert('Mínimo: 0.001 LTC'); return; } if (valor > gameData.litecoin) { alert('Saldo LTC insuficiente!'); return; } gameData.litecoin -= valor; updateDisplay(); saveGame(); closeWithdrawModal(); alert(`Saque de ${valor} LTC enviado para:\n${endereco}`); } function updateDisplay() { document.getElementById('cafecoin-balance').innerText = gameData.cafecoin.toFixed(2); document.getElementById('Litecoin-balance').innerText = gameData.litecoin.toFixed(8); document.getElementById('mining-power').innerText = gameData.miningPower.toFixed(2); document.getElementById('rack-count').innerText = gameData.machines.length; updateRacks(); } // MINERAÇÃO AUTOMÁTICA - 1 H/s = 0.00000001 LTC por segundo setInterval(() => { if (gameData.miningPower > 0) { const ganho = gameData.miningPower * 0.00000001; gameData.litecoin += ganho; updateDisplay(); saveGame(); } }, 1000); // SAVE/LOAD function saveGame() { localStorage.setItem('cafeMinerSave', JSON.stringify(gameData)); } function loadGame() { const save = localStorage.getItem('cafeMinerSave'); if (save) { gameData = JSON.parse(save); } }
